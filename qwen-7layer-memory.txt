// ---- Render stack ----
const stackEl = document.getElementById('stack');

  div.className = 'layer';

  div.onclick = () => selectLayer(l.id);
  stackEl.appendChild(div);

// ---- Render scenario buttons ----
const scenariosEl = document.getElementById('scenarios');
scenarios.forEach((s, i) => {
  const btn = document.createElement('button');
  btn.className = 'scenario-btn';
  btn.textContent = s.title;
  btn.onclick = () => runScenario(i);
  scenariosEl.appendChild(btn);
});

// ---- Select layer (show details) ----
let selectedLayerId = null;
function selectLayer(id) {
  selectedLayerId = id;
  document.querySelectorAll('.layer').forEach(el => el.classList.remove('selected'));
  document.querySelector(`.layer[data-id="${id}"]`).classList.add('selected');
  const l = layers.find(x => x.id === id);
  document.getElementById('details').innerHTML = `
    <h3>${l.id}. ${l.name}</h3>
    <div class="row"><span class="label">Timescale:</span><span class="value">${l.timescale}</span></div>
    <div class="row"><span class="label">Lives in:</span><span class="value">${l.where}</span></div>
    <div class="row"><span class="label">Stores:</span><span class="value">${l.stores}</span></div>
    <div class="row"><span class="label">Implementation:</span><span class="value">${l.impl}</span></div>
    <div class="example">💡 ${l.example}</div>
  `;
}

// ---- Run scenario ----
let currentScenarioIdx = null;
let traceTimer = null;

function runScenario(idx) {
  if (traceTimer) clearTimeout(traceTimer);
  currentScenarioIdx = idx;
  const s = scenarios[idx];

  // Highlight active button
  document.querySelectorAll('.scenario-btn').forEach((b, i) => {
    b.classList.toggle('active', i === idx);
  });
  document.getElementById('userMsg').textContent = s.msg;

  // Clear previous
  document.querySelectorAll('.layer').forEach(el => el.classList.remove('active'));
  const traceEl = document.getElementById('trace');
  traceEl.innerHTML = '';

  // Animate steps
  s.steps.forEach((step, i) => {
    traceTimer = setTimeout(() => {
      // Light up layer
      document.querySelectorAll('.layer').forEach(el => el.classList.remove('active'));
      document.querySelector(`.layer[data-id="${step.layer}"]`).classList.add('active');

      // Add trace step
      const l = layers.find(x => x.id === step.layer);
      const div = document.createElement('div');
      div.className = 'trace-step';
      div.style.background = `rgba(255,255,255,0.05)`;
      div.innerHTML = `<span class="step-num">${i+1}</span><strong>${l.short}:</strong> ${step.text}`;
      traceEl.appendChild(div);
      traceEl.scrollTop = traceEl.scrollHeight;
    }, i * 900);
  });

  // Fade out last layer after trace completes
  traceTimer = setTimeout(() => {
    document.querySelectorAll('.layer').forEach(el => el.classList.remove('active'));
  }, s.steps.length * 900 + 1500);
}

function replayScenario() {
  if (currentScenarioIdx !== null) runScenario(currentScenarioIdx);
}

How to use it
Save the code above as agent-memory.html
Open it in any modern browser (Chrome, Firefox, Safari, Edge)
Interact:
Click any layer in the stack → the right panel shows what it stores, where it lives, timescale, implementation, and a concrete example
Click a scenario (Suggest a recipe, Remember me next time, etc.) → the stack lights up layer-by-layer in activation order, and an execution trace appears on the right
Replay button re-runs the last scenario
What each scenario demonstrates
Scenario
Key memory types activated
Lesson
Suggest a recipe
Working → Parametric → Semantic → Procedural
Short session needs mostly working + procedural memory
Remember me next time
Working → Parametric → Semantic (write)
Semantic memory is the write path for user facts
Book a trip
Working → Semantic → Retrieval → Procedural
Real-world data requires retrieval memory
Improve from past runs
Working → Episodic → Semantic → Retrieval
Episodic memory enables learning from experience
Long-horizon project
All 7 layers
Complex agents need the full stack, especially prospective
Design notes
Color-coded layers — each memory type has a distinct gradient so you can visually track which is active
Sequential animation (900ms per step) — matches the real order of memory activation, not just a static list
Self-contained — no external libraries, no build step, works offline
Responsive — collapses to single column on mobile
You can extend it easily: add more scenarios by appending to the scenarios array, or add a "build order" tab showing which memory to add first (per the article's "Practical Build Order" section).
