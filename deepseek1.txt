CREATE TABLE aria_cost_telemetry (
    trace_id            UUID PRIMARY KEY,

    -- WHO (6 layers of attribution)
    user_id             TEXT NOT NULL,        -- jdoe@acme.com
    team_id             TEXT NOT NULL,        -- payments-team
    line_of_business    TEXT NOT NULL,        -- consumer-banking
    application_id      TEXT NOT NULL,        -- copilot-ide
    product_feature     TEXT NOT NULL,        -- code-completion
    cost_center         TEXT NOT NULL,        -- eng-ai-consumer
    
    -- WHAT (The actual usage)
    model_used          TEXT NOT NULL,        -- claude-3-opus-20240229
    model_family        TEXT NOT NULL,        -- claude-opus
    provider            TEXT NOT NULL,        -- anthropic
    
    -- HOW MUCH (Cost breakdown)
    input_tokens        INT NOT NULL,
    output_tokens       INT NOT NULL,
    input_cost_usd      DECIMAL(10,6) NOT NULL,   -- $0.015 per 1K
    output_cost_usd     DECIMAL(10,6) NOT NULL,   -- $0.075 per 1K
    total_cost_usd      DECIMAL(10,6) NOT NULL,
    
    -- EFFICIENCY (What we COULD have spent)
    cheapest_capable_model TEXT,                  -- claude-3-haiku
    optimal_cost_usd    DECIMAL(10,6),            -- What it should cost
    waste_usd           DECIMAL(10,6) GENERATED   -- total - optimal
        ALWAYS AS (total_cost_usd - optimal_cost_usd) VIRTUAL,
    
    -- DECISION (What ARIA did)
    aria_decision       TEXT NOT NULL,        -- ALLOW | WARN | BLOCK
    auto_optimized      BOOLEAN DEFAULT FALSE -- Did ARIA swap the model?
);

-- Real-time aggregations for FinOps dashboards:
CREATE MATERIALIZED VIEW cost_by_lob_hourly AS
SELECT 
    date_trunc('hour', timestamp) as hour,
    line_of_business,
    model_used,
    COUNT(*) as requests,
    SUM(total_cost_usd) as total_cost,
    SUM(waste_usd) as waste_detected,
    AVG(CASE WHEN auto_optimized THEN 1 ELSE 0 END) as optimization_rate
FROM aria_cost_telemetry
GROUP BY 1,2,3;
